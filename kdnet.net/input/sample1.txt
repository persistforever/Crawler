愤怒不仅是指责残暴没有人性的施暴者，也包括对那些现场围观的冷漠者。<br><br>&nbsp;&nbsp;&nbsp;&nbsp;仅仅因为有犯罪嫌疑就公然实施暴力私刑，关键的是这一幕能够在光天化日之下，能够发生在众目睽睽之中，真是令人发指！也令人痛惜。诸多可怜之人因为法盲，因为无知，因为所谓的“正义”，特别是因为没有因此而受到过处罚，于是就公然违法，公然犯罪，由此变成了面目可憎的罪犯。<br><br>&nbsp;&nbsp;&nbsp;&nbsp;也难为了警方，要不是此施暴视频曝光，估计警方会继续调查，至于处罚施暴者，估计得批评教育或者法不责众或者事出有因情有可原了。不是吗？视频曝光后，相关警方不清楚，甚至这施暴者竟然也是在9月1日自行投案自首的。<br><br>&nbsp;&nbsp;&nbsp;&nbsp;可怜的是这位28岁的孕妇，可怜的是受害者变成罪犯的村民，可悲的则是法律，是所谓依法治国的“门面”。而导致这一切的则是有法不依有法不严的执法部门与人员。正是后者的纵容与放任，才导致可怜之人的存在，导致可怜之人变成了可憎之人。 if (typeof HTMLElement != "undefined" && !HTMLElement.prototype.insertAdjacentElement) { 
    HTMLElement.prototype.insertAdjacentElement = function (where, parsedNode) { 
        switch (where) { 
            case "beforeBegin": 
                this.parentNode.insertBefore(parsedNode, this); 
                break; 
            case "afterBegin": 
                this.insertBefore(parsedNode, this.firstChild); 
                break; 
            case "beforeEnd": 
                this.appendChild(parsedNode); 
                break; 
            case "afterEnd": 
                if (this.nextSibling) { 
                    this.parentNode.insertBefore(parsedNode, this.nextSibling); 
                } else { 
                    this.parentNode.appendChild(parsedNode); 
                } 
                break; 
        } 
    }; 
    HTMLElement.prototype.insertAdjacentHTML = function (where, htmlStr) { 
        var r = this.ownerDocument.createRange(); 
        r.setStartBefore(this); 
        var parsedHTML = r.createContextualFragment(htmlStr); 
        this.insertAdjacentElement(where, parsedHTML); 
    }; 
    HTMLElement.prototype.insertAdjacentText = function (where, txtStr) { 
        var parsedText = document.createTextNode(txtStr); 
        this.insertAdjacentElement(where, parsedText);

function butsubmit(){

			this_integral = arr_radio[i].value;
		}

		}

		errmsg = "请选择或填写打赏的积分数。";
	}
	
	if (errmsg == "") {
		////是否跟帖回复
		var reply_words = $("#reply-words").val();
		var rpy_check = $("input[name='rpy_checkbox']");
		if (rpy_check[0].checked &&　reply_words == '') {
			errmsg = "您选择跟帖回复，请说点什么吧。";
		}
	}

	if (errmsg != "") {
		document.getElementById("butsubmit").attributes["class"].value = "reward-btn-big reward-btn-big2";
		alert(errmsg);
		return;
	} else {
		document.getElementById("butsubmit").attributes["onclick"].value = '';
		if (rpy_check[0].checked || (rpy_check[0].checked == false && reply_words != '')) {	//跟帖回复

			//alert(document.t_form.attributes["action"].value);
			document.t_form.submit(); 
		} else {	//不跟贴回复，直接打赏
			document.t_form.attributes["action"].value = 'http://club.kdnet.net/giving.asp?rootid=11142000&posttable=8&intetral='+this_integral+'&valid_integral='+valid_integral; 
			//alert(document.t_form.attributes["action"].value);
			document.t_form.submit(); 
		}
	}
}

			$("#notenough_memo").html('');
		}
	}
}
